import { useCallback, useEffect, useState } from "react";
import { CreditCard, Plus, RefreshCw } from "lucide-react";
import { useNavigate } from "react-router-dom";
import { subscriptionsApi } from "../api/subscriptions";
import AdminLayout from "../components/AdminLayout";
import { useToast } from "../components/useToast";
import { useAuth } from "../context/useAuth";

function formatDate(value) {
  if (!value) return "-";
  const date = new Date(value);
  return Number.isNaN(date.getTime()) ? "-" : date.toLocaleDateString();
}

function normalizeSubscriptionResponse(response) {
  if (!response) return [];
  if (Array.isArray(response)) return response;
  if (Array.isArray(response.data)) return response.data;
  if (Array.isArray(response.subscriptions)) return response.subscriptions;
  if (Array.isArray(response.payload)) return response.payload;
  return [];
}

function getSubscriptionCustomer(item) {
  if (!item) return "N/A";
  const customer = item.customer || item.user;
  if (!customer) return "N/A";
  if (typeof customer === "string") return customer;
  return customer.fullName || customer.name || customer.email || "N/A";
}

function getSubscriptionPlan(item) {
  if (!item) return "N/A";
  const plan = item.plan || item.subscription || item.service;
  if (!plan) return "N/A";
  if (typeof plan === "string") return plan;
  return plan.name || plan.title || plan.type || plan.planName || "N/A";
}

function getSubscriptionStatus(item) {
  if (!item) return "N/A";
  return (
    item.status ||
    (item.isActive === true
      ? "active"
      : item.isActive === false
        ? "inactive"
        : undefined) ||
    item.state ||
    item.statusType ||
    "N/A"
  );
}

function getSubscriptionTotal(item) {
  if (!item) return "N/A";
  if (item.totalAmount != null) return `$${item.totalAmount}`;
  if (item.amount != null) return `$${item.amount}`;
  if (item.price != null) return `$${item.price}`;
  if (item.fee != null) return `$${item.fee}`;
  return "N/A";
}

function getSubscriptionStart(item) {
  return formatDate(
    item.startDate ||
      item.start_date ||
      item.startedAt ||
      item.subscriptionStart ||
      item.activationDate,
  );
}

const emptyForm = {
  planName: "",
  billingCycle: "monthly",
  price: "",
  duration: "30",
  maxUsers: "1",
  status: "active",
  description: "",
  features: "",
};

function getPlanName(item) {
  if (!item) return "N/A";
  return (
    item.name || item.planName || item.title || item.type || item.plan || "N/A"
  );
}

function getPlanBilling(item) {
  if (!item) return "N/A";
  return (
    item.billingCycle ||
    item.billing ||
    item.period ||
    item.frequency ||
    "monthly"
  );
}

function getPlanPrice(item) {
  if (!item) return "N/A";
  const amount =
    item.price ?? item.totalAmount ?? item.amount ?? item.fee ?? item.cost;
  return amount != null ? `$${amount}` : "N/A";
}

function getPlanDescription(item) {
  if (!item) return "";
  return item.description || item.details || item.summary || "";
}

function getPlanStatus(item) {
  if (!item) return "N/A";
  return (
    item.status ||
    (item.isActive === true
      ? "active"
      : item.isActive === false
        ? "inactive"
        : undefined) ||
    item.state ||
    item.statusType ||
    "N/A"
  );
}

function Subscriptions() {
  const navigate = useNavigate();
  const { token, logout } = useAuth();
  const { showToast } = useToast();
  const [subscriptions, setSubscriptions] = useState([]);
  const [form, setForm] = useState(emptyForm);
  const [editingId, setEditingId] = useState("");
  const [isLoading, setIsLoading] = useState(true);
  const [isSaving, setIsSaving] = useState(false);
  const [backendNotice, setBackendNotice] = useState("");

  const loadData = useCallback(async () => {
    setIsLoading(true);
    setBackendNotice("");

    try {
      const response = await subscriptionsApi.list(token);
      setSubscriptions(normalizeSubscriptionResponse(response));
    } catch (err) {
      setBackendNotice(err.message || "Unable to load subscriptions.");
      showToast({
        message: err.message || "Unable to load subscriptions.",
        type: "error",
      });
      if (err.status === 401) {
        logout();
        navigate("/login", { replace: true });
      }
    } finally {
      setIsLoading(false);
    }
  }, [logout, navigate, showToast, token]);

  useEffect(() => {
    loadData();
  }, [loadData]);

  function updateForm(event) {
    const { name, value } = event.target;
    setForm((current) => ({
      ...current,
      [name]: value,
    }));
  }

  async function handleSave(event) {
    event.preventDefault();
    setIsSaving(true);
    setBackendNotice("");

    try {
      const payload = {
        name: form.planName || undefined,
        billingCycle: form.billingCycle || undefined,
        price: form.price ? Number(form.price) : undefined,
        duration: form.duration ? Number(form.duration) : undefined,
        maxUsers: form.maxUsers ? Number(form.maxUsers) : undefined,
        features: form.features
          ? form.features
              .split(",")
              .map((item) => item.trim())
              .filter(Boolean)
          : [],
        status: form.status || undefined,
        description: form.description || undefined,
      };

      const response = editingId
        ? await subscriptionsApi.update(editingId, payload, token)
        : await subscriptionsApi.create(payload, token);

      const savedSubscriptions = normalizeSubscriptionResponse(response);
      await loadData();

      showToast({
        message: editingId
          ? "Subscription plan updated successfully."
          : "Subscription plan created successfully.",
        type: "success",
      });

      setForm(emptyForm);
      setEditingId("");
      if (
        !editingId &&
        Array.isArray(savedSubscriptions) &&
        savedSubscriptions.length > 0
      ) {
        setSubscriptions((current) => [savedSubscriptions[0], ...current]);
      }
    } catch (err) {
      setBackendNotice(err.message || "Unable to save subscription plan.");
      showToast({
        message: err.message || "Unable to save subscription plan.",
        type: "error",
      });
    } finally {
      setIsSaving(false);
    }
  }

  function startEdit(item) {
    setEditingId(item._id || item.id || "");
    setForm({
      planName: getPlanName(item),
      billingCycle: getPlanBilling(item),
      price:
        item.price ??
        item.totalAmount ??
        item.amount ??
        item.fee ??
        item.cost ??
        "",
      duration: item.duration ?? item.periodDays ?? item.length ?? "30",
      maxUsers: item.maxUsers ?? item.users ?? item.seats ?? "1",
      status: getPlanStatus(item) || "active",
      description: getPlanDescription(item),
      features: Array.isArray(item.features)
        ? item.features.join(", ")
        : item.features || "",
    });
  }

  async function handleDelete(id) {
    if (!window.confirm("Are you sure you want to delete this subscription?"))
      return;

    try {
      await subscriptionsApi.delete(id, token);
      showToast({ message: "Subscription removed.", type: "success" });
      await loadData();
    } catch (err) {
      setBackendNotice(err.message || "Unable to delete subscription.");
      showToast({
        message: err.message || "Unable to delete subscription.",
        type: "error",
      });
    }
  }

  function resetForm() {
    setEditingId("");
    setForm(emptyForm);
  }

  return (
    <AdminLayout>
      <section className="mx-auto max-w-7xl">
        <div className="mb-6 flex flex-col justify-between gap-4 md:flex-row md:items-end">
          <div>
            <p className="text-sm font-medium text-app-primary">Management</p>
            <h1 className="mt-1 text-3xl font-semibold text-app-text">
              Subscription Management
            </h1>
            <p className="mt-2 max-w-2xl text-sm leading-6 text-app-muted">
              View and manage active subscriptions, plans, and billing cycles.
            </p>
          </div>
          <button
            className="inline-flex h-10 items-center gap-2 rounded-md border border-app-border px-4 text-sm text-app-text hover:bg-app-panel-muted"
            type="button"
            onClick={loadData}
            disabled={isLoading}
          >
            <RefreshCw
              className={`h-4 w-4 ${isLoading ? "animate-spin" : ""}`}
            />
            Refresh
          </button>
        </div>

        {backendNotice && (
          <div className="mb-5 rounded-md border border-red-200 bg-red-50 px-4 py-3 text-sm text-red-700">
            {backendNotice}
          </div>
        )}

        <div className="grid gap-6">
          <section className="rounded-md border border-app-border bg-app-panel p-5 shadow-sm">
            <div className="mb-5 flex flex-col gap-3 md:flex-row md:items-end md:justify-between">
              <div>
                <h2 className="text-lg font-semibold text-app-text">
                  {editingId ? "Edit Plan" : "Create General Plan"}
                </h2>
                <p className="mt-1 text-sm text-app-muted">
                  Create or update a general subscription plan available for all
                  users.
                </p>
              </div>
              {editingId ? (
                <button
                  type="button"
                  onClick={resetForm}
                  className="h-10 rounded-md border border-app-border px-4 text-sm text-app-text hover:bg-app-panel-muted"
                >
                  Cancel edit
                </button>
              ) : null}
            </div>
            <form className="grid gap-4 md:grid-cols-4" onSubmit={handleSave}>
              <label className="block">
                <span className="mb-2 block text-sm font-medium text-app-text">
                  Plan Name
                </span>
                <input
                  className="h-11 w-full rounded-md border border-app-border bg-app-input px-3 text-sm outline-none focus:border-app-primary"
                  name="planName"
                  value={form.planName}
                  onChange={updateForm}
                  placeholder="Basic, Professional, Enterprise"
                  required
                />
              </label>
              <label className="block">
                <span className="mb-2 block text-sm font-medium text-app-text">
                  Billing Cycle
                </span>
                <select
                  className="h-11 w-full rounded-md border border-app-border bg-app-input px-3 text-sm outline-none focus:border-app-primary"
                  name="billingCycle"
                  value={form.billingCycle}
                  onChange={updateForm}
                >
                  <option value="monthly">Monthly</option>
                  <option value="quarterly">Quarterly</option>
                  <option value="annual">Annual</option>
                </select>
              </label>
              <label className="block">
                <span className="mb-2 block text-sm font-medium text-app-text">
                  Price
                </span>
                <input
                  className="h-11 w-full rounded-md border border-app-border bg-app-input px-3 text-sm outline-none focus:border-app-primary"
                  type="number"
                  name="price"
                  value={form.price}
                  onChange={updateForm}
                  placeholder="0.00"
                  min="0"
                  step="0.01"
                  required
                />
              </label>
              <label className="block">
                <span className="mb-2 block text-sm font-medium text-app-text">
                  Status
                </span>
                <select
                  className="h-11 w-full rounded-md border border-app-border bg-app-input px-3 text-sm outline-none focus:border-app-primary"
                  name="status"
                  value={form.status}
                  onChange={updateForm}
                >
                  <option value="active">Active</option>
                  <option value="inactive">Inactive</option>
                  <option value="pending">Pending</option>
                </select>
              </label>
              <label className="md:col-span-4 block">
                <span className="mb-2 block text-sm font-medium text-app-text">
                  Description
                </span>
                <input
                  className="h-11 w-full rounded-md border border-app-border bg-app-input px-3 text-sm outline-none focus:border-app-primary"
                  name="description"
                  value={form.description}
                  onChange={updateForm}
                  placeholder="Plan details for all users"
                />
              </label>
              <div className="md:col-span-4 flex flex-col gap-3 sm:flex-row sm:items-center">
                <button
                  className="inline-flex h-11 items-center justify-center gap-2 rounded-md bg-app-primary px-5 text-sm font-semibold text-app-primary-text hover:bg-app-primary-hover disabled:opacity-70"
                  type="submit"
                  disabled={isSaving}
                >
                  <Plus className="h-4 w-4" />
                  {isSaving
                    ? "Saving..."
                    : editingId
                      ? "Update Plan"
                      : "Create Plan"}
                </button>
                {editingId ? (
                  <button
                    type="button"
                    onClick={resetForm}
                    className="inline-flex h-11 items-center justify-center rounded-md border border-app-border px-5 text-sm text-app-text hover:bg-app-panel-muted"
                  >
                    Clear form
                  </button>
                ) : null}
              </div>
            </form>
          </section>

          {isLoading ? (
            <div className="rounded-md border border-app-border bg-app-panel p-8 text-center shadow-sm">
              <p className="text-sm font-medium text-app-text">
                Loading subscriptions...
              </p>
            </div>
          ) : //same rule for the backend create asfrontend shows
          subscriptions.length === 0 ? (
            <div className="rounded-md border border-app-border bg-app-panel p-8 text-center shadow-sm">
              <div className="mx-auto mb-4 flex h-16 w-16 items-center justify-center rounded-full bg-app-panel-muted text-app-primary">
                <CreditCard className="h-8 w-8" />
              </div>
              <h2 className="text-xl font-semibold text-app-text">
                No Plans Found
              </h2>
              <p className="mx-auto mt-2 max-w-md text-sm text-app-muted">
                Start by creating a general subscription plan for customers.
              </p>
              <button
                className="mt-6 inline-flex h-11 items-center gap-2 rounded-md bg-app-primary px-6 text-sm font-semibold text-app-primary-text hover:bg-app-primary-hover transition-colors"
                type="button"
                onClick={resetForm}
              >
                <Plus className="h-4 w-4" />
                Create Plan
              </button>
            </div>
          ) : (
            <section className="overflow-hidden rounded-md border border-app-border bg-app-panel shadow-sm">
              <table className="w-full text-left text-sm">
                <thead className="border-b border-app-border bg-app-panel text-xs uppercase text-app-muted">
                  <tr>
                    <th className="px-5 py-3">Plan</th>
                    <th className="px-5 py-3">Billing</th>
                    <th className="px-5 py-3">Price</th>
                    <th className="px-5 py-3">Status</th>
                    <th className="px-5 py-3">Description</th>
                    <th className="px-5 py-3 text-right">Actions</th>
                  </tr>
                </thead>
                <tbody className="divide-y divide-app-border">
                  {subscriptions.map((item) => (
                    <tr key={item._id || item.id}>
                      <td className="px-5 py-4 font-medium text-app-text">
                        {getPlanName(item)}
                      </td>
                      <td className="px-5 py-4 text-app-muted capitalize">
                        {getPlanBilling(item)}
                      </td>
                      <td className="px-5 py-4 font-semibold text-app-text">
                        {getPlanPrice(item)}
                      </td>
                      <td className="px-5 py-4 text-app-muted capitalize">
                        {getPlanStatus(item)}
                      </td>
                      <td className="px-5 py-4 text-app-muted truncate max-w-60">
                        {getPlanDescription(item)}
                      </td>
                      <td className="px-5 py-4 text-right">
                        <div className="inline-flex items-center gap-2">
                          <button
                            type="button"
                            className="rounded-md border border-app-border px-3 py-2 text-xs text-app-text hover:bg-app-panel-muted"
                            onClick={() => startEdit(item)}
                          >
                            Edit
                          </button>
                          <button
                            type="button"
                            className="rounded-md border border-app-border px-3 py-2 text-xs text-app-danger hover:bg-app-danger/10"
                            onClick={() => handleDelete(item._id || item.id)}
                          >
                            Delete
                          </button>
                        </div>
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </section>
          )}

          {/* <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
            <PlanCard
              name="Basic Plan"
              price="29"
              features={[
                "Starter access",
                "Standard support",
                "Monthly billing",
              ]}
            />
            <PlanCard
              name="Professional Plan"
              price="99"
              features={[
                "Expanded seats",
                "Priority support",
                "Quarterly billing",
              ]}
              highlighted
            />
            <PlanCard
              name="Enterprise Plan"
              price="249"
              features={[
                "Unlimited seats",
                "Dedicated support",
                "Annual billing",
              ]}
            />
          </div> */}
        </div>
      </section>
    </AdminLayout>
  );
}

function PlanCard({ name, price, features, highlighted }) {
  return (
    <div
      className={`rounded-md border ${highlighted ? "border-app-primary bg-app-primary/5" : "border-app-border bg-app-panel"} p-6 shadow-sm`}
    >
      <h3 className="text-lg font-semibold text-app-text">{name}</h3>
      <div className="mt-4 flex items-baseline gap-1">
        <span className="text-3xl font-bold text-app-text">${price}</span>
        <span className="text-sm text-app-muted">/month</span>
      </div>
      <ul className="mt-6 space-y-3">
        {features.map((f) => (
          <li
            key={f}
            className="flex items-center gap-2 text-sm text-app-muted"
          >
            <div className="h-1.5 w-1.5 rounded-full bg-app-primary" />
            {f}
          </li>
        ))}
      </ul>
      <button
        className={`mt-8 w-full h-10 rounded-md text-sm font-semibold transition-colors ${
          highlighted
            ? "bg-app-primary text-app-primary-text hover:bg-app-primary-hover"
            : "border border-app-border text-app-text hover:bg-app-panel-muted"
        }`}
      >
        Manage Plan
      </button>
    </div>
  );
}

export default Subscriptions;



import { request } from "./client";

export const subscriptionsApi = {
  list(token) {
    return request("/subscriptions", { token });
  },
  get(id, token) {
    return request(`/subscriptions/${id}`, { token });
  },
  create(payload, token) {
    return request("/subscriptions", {
      method: "POST",
      body: payload,
      token,
    });
  },
  update(id, payload, token) {
    return request(`/subscriptions/${id}`, {
      method: "PATCH",
      body: payload,
      token,
    });
  },
  delete(id, token) {
    return request(`/subscriptions/${id}`, {
      method: "DELETE",
      token,
    });
  },
};
